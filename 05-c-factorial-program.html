C Program to Find Factorial

#include

int main() {
    int n, i;
    long fact = 1;

    printf("Enter a number: ");
    scanf("%d", &n);

    for(i = 1; i <= n; i++) {
        fact = fact * i;
    }

    printf("Factorial = %ld", fact);
    return 0;
}

Output

Enter a number: 5
Factorial = 120
